-- ==========================================
-- DUMP DE ESQUEMA (SOLO ESTRUCTURA)
-- ==========================================

-- TABLA: users
CREATE TABLE users (
    user_id uuid NOT NULL,
    tenant_id uuid NOT NULL,
    username character varying(100) NOT NULL,
    email character varying(255),
    password_hash character varying(255),
    auth_provider character varying(50),
    ldap_dn character varying(255),
    is_active boolean,
    last_login timestamp with time zone,
    created_at timestamp with time zone  DEFAULT now(),
    is_superadmin boolean  DEFAULT false
);

-- TABLA: cfdi_metadata
CREATE TABLE cfdi_metadata (
    uuid character varying(36) NOT NULL,
    tenant_id uuid NOT NULL,
    rfc_emisor character varying(13) NOT NULL,
    rfc_receptor character varying(13) NOT NULL,
    fecha_emision timestamp with time zone NOT NULL,
    tipo_comprobante character varying(1) NOT NULL,
    total numeric NOT NULL,
    estado character varying(20),
    tiene_pdf boolean,
    fecha_registro timestamp with time zone  DEFAULT now()
);

-- TABLA: tenants
CREATE TABLE tenants (
    tenant_id uuid NOT NULL,
    rfc character varying(13) NOT NULL,
    business_name character varying(200) NOT NULL,
    logo_path character varying(500),
    is_active boolean,
    created_at timestamp with time zone  DEFAULT now()
);

-- TABLA: bitacora_auditoria
CREATE TABLE bitacora_auditoria (
    id integer NOT NULL DEFAULT nextval('bitacora_auditoria_id_seq'::regclass),
    usuario_id uuid,
    entidad_id uuid,
    accion character varying(100) NOT NULL,
    detalle jsonb,
    fecha_hora timestamp with time zone  DEFAULT now()
);

-- TABLA: usuarios
CREATE TABLE usuarios (
    id uuid NOT NULL,
    username character varying(100) NOT NULL,
    email character varying(255) NOT NULL,
    hashed_password character varying(255),
    auth_provider character varying(50),
    is_active boolean,
    created_at timestamp with time zone  DEFAULT now(),
    is_superadmin boolean  DEFAULT false,
    last_login timestamp with time zone
);

-- TABLA: entidades_fiscales
CREATE TABLE entidades_fiscales (
    id uuid NOT NULL,
    rfc character varying(13) NOT NULL,
    razon_social character varying(255) NOT NULL,
    is_active boolean,
    created_at timestamp with time zone  DEFAULT now(),
    logo_url character varying(500)
);

-- TABLA: email_queue
CREATE TABLE email_queue (
    id integer NOT NULL DEFAULT nextval('email_queue_id_seq'::regclass),
    entidad_id uuid NOT NULL,
    cfdi_uuid character varying(36) NOT NULL,
    destinatario character varying(255) NOT NULL,
    estado character varying(20),
    intentos integer,
    proximo_intento timestamp with time zone,
    created_at timestamp with time zone  DEFAULT now()
);

-- TABLA: email_logs
CREATE TABLE email_logs (
    id integer NOT NULL DEFAULT nextval('email_logs_id_seq'::regclass),
    email_queue_id integer NOT NULL,
    respuesta_smtp text,
    fecha_intento timestamp with time zone  DEFAULT now()
);

-- TABLA: entidad_smtp_configs
CREATE TABLE entidad_smtp_configs (
    id integer NOT NULL DEFAULT nextval('entidad_smtp_configs_id_seq'::regclass),
    entidad_id uuid NOT NULL,
    hostname character varying(255),
    port integer NOT NULL,
    username character varying(255) NOT NULL,
    password_encrypted text NOT NULL,
    _use_tls_deprecated boolean,
    created_at timestamp with time zone  DEFAULT now(),
    from_address character varying(255),
    security_type character varying(50)  DEFAULT 'STARTTLS'::character varying,
    authentication_type character varying(50)  DEFAULT 'LOGIN'::character varying,
    host character varying(255)
);

-- TABLA: cfdis
CREATE TABLE cfdis (
    cfdi_id uuid NOT NULL,
    tenant_id uuid NOT NULL,
    uuid character varying(36) NOT NULL,
    rfc_emisor character varying(13) NOT NULL,
    rfc_receptor character varying(13) NOT NULL,
    issue_date timestamp with time zone NOT NULL,
    total numeric NOT NULL,
    version character varying(10) NOT NULL,
    status character varying(50),
    xml_file_path character varying(1000) NOT NULL,
    pdf_file_path character varying(1000),
    created_at timestamp with time zone  DEFAULT now(),
    source_type character varying(50)  DEFAULT 'UPLOAD_MANUAL'::character varying,
    tipo_comprobante character(1),
    serie character varying(25),
    folio character varying(40),
    metadata_xml jsonb,
    metodo_pago character varying(5),
    forma_pago character varying(5),
    parent_uuid character varying(36)
);

-- TABLA: tenant_storage_configs
CREATE TABLE tenant_storage_configs (
    id integer NOT NULL DEFAULT nextval('tenant_storage_configs_id_seq'::regclass),
    tenant_id uuid NOT NULL,
    source_type character varying(50) NOT NULL,
    base_path character varying(500) NOT NULL
);

-- TABLA: comprobantes
CREATE TABLE comprobantes (
    uuid uuid NOT NULL,
    serie character varying(20),
    folio character varying(50),
    rfc_emisor character varying(13),
    rfc_receptor character varying(13),
    version character varying(5),
    total numeric,
    ruta_resguardo text,
    fecha_emision timestamp without time zone,
    ingestado_at timestamp without time zone,
    entidad_id uuid,
    tipo_comprobante character varying,
    metodo_pago character varying,
    forma_pago character varying,
    status character varying  DEFAULT 'VALID'::character varying,
    moneda character varying(10)  DEFAULT 'MXN'::character varying,
    tipo_cambio numeric,
    descuento numeric,
    total_impuestos_trasladados numeric,
    total_impuestos_retenidos numeric,
    subtotal numeric  DEFAULT 0,
    nombre_emisor character varying(255),
    nombre_receptor character varying(255),
    estatus_sat character varying(20)
);

-- TABLA: cfdi_relacionados
CREATE TABLE cfdi_relacionados (
    id integer NOT NULL DEFAULT nextval('cfdi_relacionados_id_seq'::regclass),
    cfdi_id uuid NOT NULL,
    uuid_padre character varying(36) NOT NULL,
    uuid_relacionado character varying(36) NOT NULL,
    tipo_relacion character varying(10),
    monto_pagado numeric,
    saldo_insoluto numeric,
    num_parcialidad integer
);

-- TABLA: entidad_storage_configs
CREATE TABLE entidad_storage_configs (
    id integer NOT NULL DEFAULT nextval('entidad_storage_configs_id_seq'::regclass),
    entidad_id uuid NOT NULL,
    source_type character varying(50) NOT NULL,
    base_path character varying(500) NOT NULL
);

-- TABLA: usuario_entidad_acceso
CREATE TABLE usuario_entidad_acceso (
    usuario_id uuid NOT NULL,
    entidad_id uuid NOT NULL,
    rol character varying(50) NOT NULL,
    fecha_asignacion timestamp with time zone  DEFAULT now()
);

-- TABLA: plantillas_correo
CREATE TABLE plantillas_correo (
    id integer NOT NULL DEFAULT nextval('plantillas_correo_id_seq'::regclass),
    entidad_id uuid,
    tipo_plantilla character varying(50) NOT NULL,
    asunto character varying(255) NOT NULL,
    cuerpo_html text NOT NULL,
    created_at timestamp with time zone  DEFAULT now()
);
